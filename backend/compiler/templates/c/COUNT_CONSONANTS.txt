#include <stdio.h>
#include <ctype.h>
#include <string.h>

int main() {
    char s[100];
    int count = 0;
    printf("Enter string: ");
    scanf("%s", s);
    for(int i=0; i<strlen(s); i++) {
        char c = tolower(s[i]);
        if(c >= 'a' && c <= 'z' && !(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u'))
            count++;
    }
    printf("Consonants: %d", count);
    return 0;
}
