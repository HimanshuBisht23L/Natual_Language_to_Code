#include<stdio.h>
#include<limits.h>

void Swap(int *A, int *B){
	int temp = *A;
	*A = *B;
	*B = temp;
}

void BubbleSort(int n, int arr[n]){
	int chk = 0;
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n - i - 1; j++)
		{
			if (arr[j] > arr[j+1])
			{
				chk = 1;
				Swap(&arr[j], &arr[j+1]);
			}
		}
		if (!chk)
		{
			break;
		}
		
		
	}
	
}


int main(){
	int arr[] = {20, 10, 5, 30, 2, 15};
	BubbleSort(6, arr);

	for (int i = 0; i < 6; i++)
	{
		printf("%d ", arr[i]);
	}
	
	return 0;
}
