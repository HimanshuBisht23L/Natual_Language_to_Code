#include<stdio.h>

int main()
{

    int arr[5] = {10,20,30,40,50};

    int key;


    printf("Enter element: ");

    scanf("%d", &key);


    for(int i=0;i<5;i++)
    {

        if(arr[i]==key)
        {

            printf("Found at index %d", i);

            return 0;

        }

    }


    printf("Not found");

    return 0;

}
